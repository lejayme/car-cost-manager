四、核心功能实现

4.1 添加开销 (expense_provider.dart)

`dart import 'package:flutter_riverpod/flutter_riverpod.dart'; import 'database.dart'; import 'expense.dart';

final expenseProvider = StateNotifierProvider>((ref) { return ExpenseNotifier(); });

class ExpenseNotifier extends StateNotifier> { ExpenseNotifier() : super([]) { loadExpenses(); } Future loadExpenses() async { final expenses = await DatabaseService.isar.expenses .filter() .dateGreaterThan(DateTime.now().subtract(const Duration(days: 30))) .sortByDate() .findAll(); state = expenses; } Future addExpense(Expense expense) async { await DatabaseService.isar.writeTxn(() => DatabaseService.isar.expenses.put(expense) ); loadExpenses(); } Future deleteExpense(int id) async { await DatabaseService.isar.writeTxn(() => DatabaseService.isar.expenses.delete(id) ); loadExpenses(); } } `

4.2 统计服务 (stats_service.dart)

`dart import 'database.dart'; import 'expense.dart';

class StatsService { // 今日总支出 static Future getTodayTotal() async { final now = DateTime.now(); final start = DateTime(now.year, now.month, now.day); final expenses = await DatabaseService.isar.expenses .filter() .dateGreaterOrEqual(start) .findAll(); return expenses.fold(0.0, (sum, e) => sum + e.amount); } // 本月总支出 static Future getMonthTotal() async { final now = DateTime.now(); final start = DateTime(now.year, now.month, 1); final expenses = await DatabaseService.isar.expenses .filter() .dateGreaterOrEqual(start) .findAll(); return expenses.fold(0.0, (sum, e) => sum + e.amount); } // 日均成本 static Future getDailyAverage({int days = 30}) async { final start = DateTime.now().subtract(Duration(days: days)); final expenses = await DatabaseService.isar.expenses .filter() .dateGreaterOrEqual(start) .findAll(); final total = expenses.fold(0.0, (sum, e) => sum + e.amount); return total / days; } // 分类统计 static Future> getCategoryStats({int days = 30}) async { final start = DateTime.now().subtract(Duration(days: days)); final expenses = await DatabaseService.isar.expenses .filter() .dateGreaterOrEqual(start) .findAll(); final Map stats = {}; for (final expense in expenses) { stats[expense.categoryId] = (stats[expense.categoryId] ?? 0) + expense.amount; } return stats; } } `
